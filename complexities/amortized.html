Szczegółowa złożoność

Amortized O(1)

Amortized time oznacza, że raz na jakiś czas trafia się droga operacja, ale średni koszt na jedną operację pozostaje niski. Klasyczny przykład to append w dynamicznej tablicy.

Wróć do przeglądu

Złożoności / Amortized O(1)

Najkrótsza intuicja

Rzadki koszt rozmywa się w średniej

Jeśli masz dużo tanich operacji i od czasu do czasu jedną drogą, całkowity koszt może nadal wyjść liniowo, a średni koszt jednej operacji stały. Tak działa między innymi append w dynamic array, np. C++ vector.push_back().

Typowe przykłady

  • append w dynamicznej tablicy
  • append w dynamic array, np. C++ vector.push_back()
  • powiększanie bufora przez podwojenie pojemności

Skąd to się bierze

Resize jest kosztowny, bo trzeba skopiować stare dane. Ale zdarza się tak rzadko, że średnio pojedyncze dopisanie nadal wychodzi O(1).

Kluczowy takeaway

Amortized time to średni koszt na operację, a nie koszt najdroższej pojedynczej operacji.

Przykład: własna dynamiczna tablica w C
#include <stdio.h>
#include <stdlib.h>

typedef struct {
    int* data;
    int size;
    int capacity;
} DynamicArray;

void append(DynamicArray* arr, int value) {
    if (arr->size == arr->capacity) {
        int new_capacity = arr->capacity == 0 ? 1 : arr->capacity * 2;
        int* new_data = malloc(sizeof(int) * new_capacity);

        for (int i = 0; i < arr->size; i++) {
            new_data[i] = arr->data[i];
        }

        free(arr->data);
        arr->data = new_data;
        arr->capacity = new_capacity;
    }

    arr->data[arr->size++] = value;
}

int main(void) {
    DynamicArray arr = {NULL, 0, 0};
    append(&arr, 10);
    append(&arr, 20);
    append(&arr, 30);

    printf("%d %d %d\n", arr.data[0], arr.data[1], arr.data[2]);
    free(arr.data);
    return 0;
}
Przykład: podwajanie pojemności
#include <algorithm>
#include <iostream>
#include <memory>

struct DynamicArray {
    std::unique_ptr<int[]> data;
    int size = 0;
    int capacity = 0;

    void append(int value) {
        if (size == capacity) {
            int new_capacity = capacity == 0 ? 1 : capacity * 2;
            std::unique_ptr<int[]> bigger(new int[new_capacity]);

            for (int i = 0; i < size; ++i) {
                bigger[i] = data[i];
            }

            data = std::move(bigger);
            capacity = new_capacity;
        }

        data[size++] = value;
    }
};

int main() {
    DynamicArray arr;
    arr.append(10);
    arr.append(20);
    arr.append(30);

    std::cout << arr.data[0] << ' ' << arr.data[1] << ' ' << arr.data[2] << '\n';
    return 0;
}
Przykład: append z okazjonalnym resize
using System;

class DynamicArray
{
    private int[] data = new int[1];
    private int size = 0;

    public void Append(int value)
    {
        if (size == data.Length)
        {
            int[] bigger = new int[data.Length * 2];
            Array.Copy(data, bigger, size);
            data = bigger;
        }

        data[size++] = value;
    }

    public int this[int index] => data[index];
}

class Program
{
    static void Main()
    {
        DynamicArray arr = new DynamicArray();
        arr.Append(10);
        arr.Append(20);
        arr.Append(30);

        Console.WriteLine($"{arr[0]} {arr[1]} {arr[2]}");
    }
}
Przykład: własny append z powiększaniem bufora
class DynamicArray:
    def __init__(self) -> None:
        self.capacity = 1
        self.size = 0
        self.data = [None] * self.capacity

    def append(self, value: int) -> None:
        if self.size == self.capacity:
            bigger = [None] * (self.capacity * 2)
            for i in range(self.size):
                bigger[i] = self.data[i]
            self.data = bigger
            self.capacity *= 2

        self.data[self.size] = value
        self.size += 1


arr = DynamicArray()
arr.append(10)
arr.append(20)
arr.append(30)
print(arr.data[:arr.size])
